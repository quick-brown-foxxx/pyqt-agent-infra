# src/qt_ai_dev_tools/subsystems/_subprocess.py
"""Typed subprocess wrapper for system tool calls."""

from __future__ import annotations

import shutil
import subprocess
from pathlib import Path


def check_tool(name: str) -> Path:
    """Find system tool or raise RuntimeError with install hint."""
    path = shutil.which(name)
    if path is None:
        msg = f"Required tool '{name}' not found. Install it in the VM provision."
        raise RuntimeError(msg)
    return Path(path)


def run_tool(
    args: list[str],
    *,
    input_data: str | None = None,
    timeout: float = 30.0,
    env: dict[str, str] | None = None,
) -> str:
    """Run a system tool, return stdout. Raises RuntimeError on failure."""
    result = subprocess.run(
        args,
        capture_output=True,
        text=True,
        input=input_data,
        timeout=timeout,
        env=env,
    )
    if result.returncode != 0:
        msg = f"Tool failed: {' '.join(args)}\nstderr: {result.stderr.strip()}"
        raise RuntimeError(msg)
    return result.stdout
```

"""Clipboard access via xclip (X11)."""

from __future__ import annotations

import os

from qt_ai_dev_tools.subsystems._subprocess import check_tool, run_tool


def _clipboard_env() -> dict[str, str]:
    """Environment with DISPLAY set for xclip."""
    env = os.environ.copy()
    env.setdefault("DISPLAY", ":99")
    return env


def write(text: str) -> None:
    """Write text to X11 clipboard. Raises RuntimeError if xclip missing."""
    check_tool("xclip")
    run_tool(
        ["xclip", "-selection", "clipboard"],
        input_data=text,
        env=_clipboard_env(),
    )


def read() -> str:
    """Read text from X11 clipboard. Raises RuntimeError if xclip missing."""
    check_tool("xclip")
    return run_tool(
        ["xclip", "-selection", "clipboard", "-o"],
        env=_clipboard_env(),
    )

        self.open_btn.setAccessibleName("open_file")

        layout.addWidget(self.open_btn)

        self.save_btn = QPushButton("Save As")
        self.save_btn.setAccessibleName("save_as")
        self.save_btn.clicked.connect(self._save_file)
        layout.addWidget(self.save_btn)

        self.status_label = QLabel("No file loaded")
        self.status_label.setAccessibleName("loaded_file")
        layout.addWidget(self.status_label)

        self.text_edit = QTextEdit()
        self.text_edit.setAccessibleName("content")
        layout.addWidget(self.text_edit)

    def _open_file(self) -> None:
        path, _ = QFileDialog.getOpenFileName(self, "Open File")
        if path:
            with open(path) as f:
                self.text_edit.setPlainText(f.read())
            self.status_label.setText(path)

    def _save_file(self) -> None:
        path, _ = QFileDialog.getSaveFileName(self, "Save File")
        if path:
            with open(path, "w") as f:
                f.write(self.text_edit.toPlainText())
            self.status_label.setText(f"Saved: {path}")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    from qt_ai_dev_tools.bridge import start
    start()
    window = FileDialogTestApp()
    window.show()
    sys.exit(app.exec())

        central = QWidget()

        layout.addWidget(notify_btn)

        tray_menu = QMenu()

        show_action.triggered.connect(self._show_window)

"""PipeWire audio: virtual mic, record, play, verify via sox."""

from __future__ import annotations

import os
import re
import signal
import subprocess
from pathlib import Path

from qt_ai_dev_tools.subsystems._subprocess import check_tool, run_tool
from qt_ai_dev_tools.subsystems.models import (
    AudioSource,
    AudioStream,
    AudioVerification,
    VirtualMicInfo,
)

_virtual_mic_pid: int | None = None


def virtual_mic_start(node_name: str = "virtual-mic") -> VirtualMicInfo:
    """Start PipeWire virtual mic via pw-loopback. Returns info with PID."""
    global _virtual_mic_pid
    check_tool("pw-loopback")
    proc = subprocess.Popen(
        [
            "pw-loopback",
            f"--capture-props=media.class=Audio/Source/Virtual node.name={node_name} "
            f'node.description="Virtual Microphone"',
        ],
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
    )
    _virtual_mic_pid = proc.pid
    return VirtualMicInfo(pid=proc.pid, node_name=node_name)


def virtual_mic_stop() -> None:
    """Stop the virtual mic process."""
    global _virtual_mic_pid
    if _virtual_mic_pid is not None:
        os.kill(_virtual_mic_pid, signal.SIGTERM)
        _virtual_mic_pid = None


def virtual_mic_play(path: Path, node_name: str = "virtual-mic") -> None:
    """Play audio file into virtual mic. Converts to WAV if needed."""
    check_tool("pw-cat")
    wav_path = path
    if path.suffix.lower() != ".wav":
        check_tool("ffmpeg")
        wav_path = path.with_suffix(".wav")
        run_tool(["ffmpeg", "-y", "-i", str(path), "-f", "wav", "-ar", "48000", "-ac", "1", str(wav_path)])
    run_tool(["pw-cat", "--playback", f"--target={node_name}", str(wav_path)])

        ["sox", str(path), "-n", "stat"],
        capture_output=True,
        text=True,
        timeout=30,
    )
    stderr = result.stderr

    max_amp = 0.0
    rms_amp = 0.0
    duration = 0.0

    for line in stderr.splitlines():
        if "Maximum amplitude" in line:
            max_amp = float(line.split(":")[1].strip())
        elif "RMS     amplitude" in line:
            rms_amp = float(line.split(":")[1].strip())
        elif "Length (seconds)" in line:
            duration = float(line.split(":")[1].strip())

    return AudioVerification(
        is_silent=max_amp < threshold,
        max_amplitude=max_amp,
        rms_amplitude=rms_amp,
        duration_seconds=duration,
    )
```
